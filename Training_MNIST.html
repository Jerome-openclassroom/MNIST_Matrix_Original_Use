def plot_multiple_images(x_data, y_data, indices, columns=12):
    rows = len(indices) // columns + int(len(indices) % columns > 0)
    plt.figure(figsize=(columns, rows * 1.5))
    for i, idx in enumerate(indices):
        plt.subplot(rows, columns, i + 1)
        plt.imshow(x_data[idx].reshape(28,28), cmap='gray')
        plt.title(str(y_data[idx]), fontsize=10)
        plt.axis('off')
    plt.tight_layout()
    plt.show()

# ------------------------------------------------------------------------
#  Commentaire pdagogique :
# Ici, nous adoptons une perspective gomtrique sur la matrice de confusion.
# Chaque cellule (i,j) est interprte comme un point pondr (erreur i  j),
# o la pondration est le nombre doccurrences de cette erreur.
#
# La diagonale (i = j) reprsente la prdiction parfaite. Les erreurs sont
# donc vues comme une "dispersion" autour de cette diagonale idale.
#
# Nous calculons :
# - la distance moyenne aux prdictions parfaites (MAE)
# - l'erreur quadratique moyenne (MSE), qui pse plus les confusions graves
# - et la masse hors-diagonale (off-diagonal mass), reprsentant lintensit
#   globale de la dispersion hors de la justesse.
#
# Ce glissement conceptuel transforme une matrice discrte en un nuage 
# d'erreurs pondres, introduisant une lecture statistique continue de la 
# qualit de classification.
# ------------------------------------------------------------------------


# Initialisation
n_classes = cm.shape[0]
errors = []
weights = []

# On parcourt toute la matrice cm[i,j]
for i in range(n_classes):
    for j in range(n_classes):
        count = cm[i, j]
        if count == 0:
            continue
        dist = abs(i - j)                   # Distance  la diagonale (|i - j|)
        sq_dist = (i - j)**2                # Distance au carr
        errors.append((dist, sq_dist, count))
        weights.append(count)

# Extraction des colonnes
dists     = np.array([e[0] for e in errors])
sq_dists  = np.array([e[1] for e in errors])
w         = np.array([e[2] for e in errors])

# Calculs
mean_abs_error  = np.sum(dists * w) / np.sum(w)
mean_sq_error   = np.sum(sq_dists * w) / np.sum(w)
off_diag_mass   = (np.sum(w) - np.trace(cm)) / np.sum(w)

print(f"Mean Absolute Error (MAE)       : {mean_abs_error:.4f}")
print(f"Mean Squared Error (MSE)        : {mean_sq_error:.4f}")
print(f"Off-diagonal mass (error ratio) : {off_diag_mass:.4%}")

# ------------------------------------------------------------------------
#  Interprtation avance des mesures :
#
# - Le MAE (Mean Absolute Error) reprsente ici une **distance moyenne  la diagonale**,
#   cest--dire  la prdiction parfaite. Plus MAE est faible, plus les erreurs sont
#   "proches" de la vrit. Cela prend tout son sens dans un contexte o les classes
#   ont une **structure ordonne ou continue**, comme des niveaux de ressemblance
#   (ex. : visages humains classs de 1  10, deux personnes affectes de chiffres proches (2 pour l'une, 3 pour l'autre) se ressemblent comme des jumeaux).
#
# - Le MSE (Mean Squared Error), en amplifiant les erreurs lointaines, permet de
#   quantifier la **gravit** des confusions : une erreur de 1  2 est bnigne,
#   mais une confusion de 1  9 est svrement pnalise.
#
# - L'Off-diagonal mass exprime la **proportion derreurs**,
#   soit 1 - accuracy. Elle donne une **mesure brute du taux de dispersion**,
#   indpendamment de la gravit ou de la proximit des erreurs.
#
# Cette approche permet de dpasser la simple "prcision moyenne" pour entrer
# dans une **lecture gomtrique, smantique et nuance** des erreurs dun modle,
# particulirement utile lorsque les classes portent une continuit (valeurs, visages,
# stades biologiques, etc.).
# ------------------------------------------------------------------------
